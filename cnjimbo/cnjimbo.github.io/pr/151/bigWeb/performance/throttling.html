function throttle(fn, delay) {
    let start = 0, timer = null
    return function () {
        let end = Date.now()
        let context = this;
        let args = arguments
        if (end - start < delay) {
            if (timer) {
                clearTimeout(timer)
            }
            timer = setTimeout(() => {
                start = end
                fn.apply(context, [...args])
            }, delay)
        } else {
            start = end
            fn.call(context, ...args)
            clearTimeout(timer)
        }
    }
}
